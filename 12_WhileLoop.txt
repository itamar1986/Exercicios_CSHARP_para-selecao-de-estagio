using System;
public class WhileLoop
{
	public static void Main()
	{
		int myInt = 0;
		while (myInt < 10)
		{
		Console.Write("{0} ", myInt);
		myInt++;
		}
		Console.WriteLine();
	}
}
/*
while A sintaxe desta condicional é:
while (<condição>)
{
	...
	<expressões>
	...
}
*/
